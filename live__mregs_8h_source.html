30 class LiveMregsAnalysis : public BackwardAnalysis {
31 public:
32  // There are only 16 mregs
33  static const unsigned MAX_MREGS = 16;
34 
35  // Fact type is bitset of machine register numbers
36  typedef std::bitset<MAX_MREGS> FactType;
37 
38  // The "top" fact is an unknown value that combines nondestructively
39  // with known facts. For this analysis, it's the empty set.
40  FactType get_top_fact() const { return FactType(); }
41 
42  // Combine live sets. For this analysis, we use union.
43  FactType combine_facts(const FactType &left, const FactType &right) const {
44  return left | right;
45  }
46 
47  // Model an instruction.
48  void model_instruction(Instruction *ins, FactType &fact) const {
49  // Model an instruction (backwards). If the instruction is a def,
50  // the assigned-to mreg is killed. Every mreg used in the instruction,
51  // the mreg becomes alive (or is kept alive.)
52 
53  if (LowLevel::is_def(ins)) {
54  std::vector<MachineReg> defs = LowLevel::get_def_mregs(ins);
55  for (auto i = defs.begin(); i != defs.end(); ++i)
56  fact.reset(unsigned(*i));
57  }
58 
59  std::vector<MachineReg> uses = LowLevel::get_use_mregs(ins);
60  for (auto i = uses.begin(); i != uses.end(); ++i)
61  fact.set(unsigned(*i));
62  }
63 
64  // Convert a dataflow fact to a string (for printing the CFG annotated with
65  // dataflow facts)
66  std::string fact_to_string(const FactType &fact) const {
67  LowLevelFormatter ll_formatter;
68 
69  std::string s("{");
70  for (unsigned i = 0; i < MAX_MREGS; i++) {
71  if (fact.test(i)) {
72  if (s != "{") { s += ","; }
73  // Machine registers are shown using their 64-bit name
74  Operand mreg_operand(Operand::MREG64, i);
75  s += ll_formatter.format_operand(mreg_operand);
76  }
77  }
78  s += "}";
79  return s;
80  }
81 };
