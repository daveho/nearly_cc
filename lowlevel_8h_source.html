31 // Enumeration for naming machine registers.
32 // Note that these names correspond to the full 64-bit name
33 // of each register. However, when used as the register number
34 // in an Operand, the Operand kind will select whether the
35 // 64-bit (Operand::MREG64), 32-bit (Operand::MREG32),
36 // 16-bit (Operand::MREG16), or 8-bit (Operand::MREG8)
37 // variant of the register is being accessed.

144 // LowLevelInstructionProperties class:
145 // this is used by ControlFlowGraphBuilder to make sense of
146 // control flow involving low-level instructions
147 // control flow involving high-level instructions
148 class LowLevelInstructionProperties {
149 public:
150  bool is_function_call(Instruction *ins) const {
151  return ins->get_opcode() == MINS_CALL;
152  }
153 
154  bool falls_through(Instruction *ins) const {
155  // only an unconditional jump instruction does not fall through
156  return ins->get_opcode() != MINS_JMP;
157  }
